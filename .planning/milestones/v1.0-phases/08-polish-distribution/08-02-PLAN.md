Create `src-ui/src/hooks/useKeyboardShortcuts.ts`:

A custom hook that registers a `keydown` event listener on `window` with these shortcuts:

| Shortcut | Action | Implementation |
|----------|--------|----------------|
| Ctrl+R or F5 | Refresh branches | Call `useBranchStore.getState().fetchBranches(projectPath)` where projectPath comes from `useConfigStore.getState()` selected project. Prevent default (Ctrl+R would reload webview). |
| Ctrl+N | New worktree | Dispatch a custom event `grove:new-worktree` that Dashboard listens for to open NewWorktreeDialog. Prevent default. |
| Ctrl+, (comma) | Open settings | Call `useConfigStore.getState().setActiveView('settings')`. Prevent default. |
| Escape | Close dialog | Dispatch custom event `grove:close-dialog`. Individual dialogs can listen for this. |

Use `useEffect` with cleanup. Use `getState()` for store access (not subscriptions) to avoid stale closures. The hook takes no arguments and returns nothing.

```typescript
import { useEffect } from 'react';
import { useConfigStore } from '@/stores/config-store';
import { useBranchStore } from '@/stores/branch-store';

export function useKeyboardShortcuts() {
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      // Ctrl+R or F5: refresh
      if ((e.ctrlKey && e.key === 'r') || e.key === 'F5') {
        e.preventDefault();
        const state = useConfigStore.getState();
        const project = state.config?.projects?.find(
          (p: any) => p.id === state.selectedProjectId
        );
        if (project) {
          useBranchStore.getState().fetchBranches(project.path);
        }
      }
      // Ctrl+N: new worktree
      if (e.ctrlKey && e.key === 'n') {
        e.preventDefault();
        window.dispatchEvent(new CustomEvent('grove:new-worktree'));
      }
      // Ctrl+,: settings
      if (e.ctrlKey && e.key === ',') {
        e.preventDefault();
        useConfigStore.getState().setActiveView('settings');
      }
      // Escape: close dialog
      if (e.key === 'Escape') {
        window.dispatchEvent(new CustomEvent('grove:close-dialog'));
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, []);
}
```
