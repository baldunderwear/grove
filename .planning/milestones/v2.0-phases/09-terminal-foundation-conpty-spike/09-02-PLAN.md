```typescript
import { invoke, Channel } from '@tauri-apps/api/core';

type TerminalEvent =
  | { type: 'Data'; data: string }
  | { type: 'Exit'; code: number | null }
  | { type: 'Error'; message: string };

const onEvent = new Channel<TerminalEvent>();
onEvent.onmessage = (event) => { /* handle event */ };

const terminalId = await invoke<string>('terminal_spawn', {
  workingDir: path,
  cols: terminal.cols,
  rows: terminal.rows,
  onEvent,
});

await invoke('terminal_write', { terminalId, data: 'hello' });
await invoke('terminal_resize', { terminalId, cols: 120, rows: 40 });
await invoke('terminal_kill', { terminalId });
```
